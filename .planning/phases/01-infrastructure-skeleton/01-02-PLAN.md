```python
from functools import lru_cache

from pydantic import Field
from pydantic_settings import BaseSettings, SettingsConfigDict


class Settings(BaseSettings):
    """
    Конфигурация приложения. Читается из .env (или переменных окружения).
    Все поля обязательны к заданию (пустых значений нет) кроме neo4j_password.
    """

    # Graph DB (per D-07)
    neo4j_uri: str = Field(default="bolt://localhost:7687", description="Neo4j Bolt URI")
    neo4j_user: str = Field(default="neo4j", description="Neo4j username")
    neo4j_password: str = Field(description="Neo4j password — обязательно в .env")

    # Vector store
    qdrant_url: str = Field(default="http://localhost:6333", description="Qdrant REST URL")

    # Task queue
    redis_url: str = Field(default="redis://localhost:6379", description="Redis connection URL")

    # Logging (per D-07, D-12, D-13)
    log_level: str = Field(default="INFO", description="Loguru log level: TRACE/DEBUG/INFO/WARNING/ERROR")
    log_json: bool = Field(default=False, description="LOG_JSON=true → serialize=True в loguru")

    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        case_sensitive=False,  # LOG_LEVEL и log_level — одно и то же
    )


@lru_cache
def get_settings() -> Settings:
    """
    Возвращает кэшированный экземпляр Settings (per D-08).
    Кэш сбрасывается через get_settings.cache_clear() в тестах.
    """
    return Settings()
```

Детали реализации:
- pydantic-settings v2: использовать `SettingsConfigDict`, НЕ `class Config`.
- `neo4j_password` без default — pydantic-settings поднимет ValidationError если не задан в .env.
- `lru_cache` на `get_settings()` — один экземпляр на процесс (per D-08).
- `case_sensitive=False` — `LOG_LEVEL=DEBUG` и `log_level=DEBUG` работают одинаково.
- Docker-сеть: `neo4j://neo4j:7687` (внутри), `bolt://localhost:7687` (дефолт для локального запуска).

```python
import sys

from loguru import logger


def setup_logging(level: str = "INFO", json_mode: bool = False) -> None:
    """
    Настраивает loguru global singleton (per D-09, D-10).
    Вызывается ОДИН РАЗ в lifespan. Все остальные модули делают только:
        from loguru import logger
    и используют глобальный singleton без дополнительной настройки.

    Args:
        level: Уровень логирования (TRACE/DEBUG/INFO/WARNING/ERROR) — per D-12.
        json_mode: True → serialize=True (JSON в stderr) — per D-13.
                   False → цветной формат в stderr — per D-11.
    """
    # Удалить все существующие handlers (per D-10)
    logger.remove()

    if json_mode:
        # JSON-режим: LOG_JSON=true в .env → serialize=True (per D-13)
        logger.add(
            sys.stderr,
            level=level,
            serialize=True,
        )
    else:
        # Цветной консольный вывод (per D-11)
        # Точный формат из D-11: "{time:HH:mm:ss} | {level:<8} | {name}:{line} | {message}"
        logger.add(
            sys.stderr,
            level=level,
            format="{time:HH:mm:ss} | {level:<8} | {name}:{line} | {message}",
            colorize=True,
        )
```
