```python
import asyncio
from contextlib import asynccontextmanager
from typing import AsyncGenerator

from loguru import logger
from neo4j import AsyncGraphDatabase, AsyncDriver, AsyncSession


class GraphDB:
    """
    Async Neo4j driver wrapper (per D-05).

    Предоставляет:
    - ping(): проверка связи с Neo4j (RETURN 1)
    - session(): async context manager для выполнения запросов
    - connect_with_retry(): startup с graceful degradation (per D-04)
    - close(): корректное закрытие driver
    """

    def __init__(self, uri: str, user: str, password: str) -> None:
        self._uri = uri
        self._user = user
        self._password = password
        self._driver: AsyncDriver | None = None
        self.is_connected: bool = False

    async def connect_with_retry(
        self,
        retries: int = 3,
        delays: list[int] | None = None,
    ) -> None:
        """
        Инициализирует async driver и проверяет связь с Neo4j (per D-04).
        Retry-последовательность: 1s → 3s → 9s.
        После исчерпания попыток: is_connected=False, не raise (graceful degradation).
        """
        if delays is None:
            delays = [1, 3, 9]

        self._driver = AsyncGraphDatabase.driver(
            self._uri,
            auth=(self._user, self._password),
        )

        for attempt in range(1, retries + 1):
            try:
                await self.ping()
                self.is_connected = True
                logger.info("Neo4j connected on attempt {}/{}", attempt, retries)
                return
            except Exception as exc:
                logger.warning(
                    "Neo4j ping failed (attempt {}/{}): {}", attempt, retries, exc
                )
                if attempt < retries:
                    delay = delays[attempt - 1] if attempt - 1 < len(delays) else delays[-1]
                    logger.info("Retrying in {}s...", delay)
                    await asyncio.sleep(delay)

        # После всех попыток — graceful degradation (per D-04)
        logger.error(
            "Neo4j unavailable after {} attempts — running in degraded mode", retries
        )
        self.is_connected = False

    async def ping(self) -> None:
        """
        Проверяет связь: выполняет RETURN 1.
        Raise если driver не инициализирован или Neo4j недоступен.
        """
        if self._driver is None:
            raise RuntimeError("GraphDB driver not initialized — call connect_with_retry first")
        async with self._driver.session() as session:
            await session.run("RETURN 1")

    @asynccontextmanager
    async def session(self) -> AsyncGenerator[AsyncSession, None]:
        """
        Async context manager для сессий (per D-05): async with db.session() as s.
        Raise если is_connected=False.
        """
        if self._driver is None or not self.is_connected:
            raise RuntimeError("Neo4j is not connected — cannot create session")
        async with self._driver.session() as neo4j_session:
            yield neo4j_session

    async def close(self) -> None:
        """Корректно закрывает async driver. Вызывается в lifespan shutdown."""
        if self._driver is not None:
            await self._driver.close()
            self._driver = None
            self.is_connected = False
            logger.info("Neo4j driver closed")
```

Детали реализации:
- `AsyncGraphDatabase.driver` — асинхронный API neo4j>=5.x (не `GraphDatabase.driver`).
- `connect_with_retry` использует `asyncio.sleep` (не tenacity, не threading) для простоты.
- `session()` — `@asynccontextmanager`, возвращает `AsyncSession` из neo4j.
- `is_connected` проверяется в `session()` — защита от вызова при degraded.
- `from loguru import logger` — глобальный singleton (per D-14), без дополнительной настройки.
- Нет импорта стандартного `logging`.

```dockerfile
FROM python:3.11-slim

# Non-root user (security best practice — per threat model)
RUN addgroup --system app && adduser --system --ingroup app app

WORKDIR /app

# Установить зависимости отдельным слоем (кэширование)

RUN pip install --no-cache-dir -e ".[dev]"

# Скопировать исходники (в prod; в dev — перекрывается volume mount)

USER app

# Health-check: /health всегда 200 (per D-06)

    CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health')"

EXPOSE 8000

CMD ["uvicorn", "api.main:app", "--host", "0.0.0.0", "--port", "8000"]
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Docker network → Neo4j | GraphDB подключается к neo4j:7687 внутри Docker-сети |
| Host → Docker container | .env монтируется в контейнер через env_file |
| Container filesystem → host | Volume mount .:/app открывает весь код хоста в контейнере |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Information Disclosure | .env попадает в git через `git add .` | mitigate | `.env` добавлен в `.gitignore`; `!.env.example` сохраняет шаблон в репозитории |
| T-03-02 | Elevation of Privilege | Docker container запускается от root | mitigate | Dockerfile создаёт `app` user (adduser --system) и переключает на него через `USER app` |
| T-03-03 | Tampering | Volume mount `.:/app` даёт контейнеру write-доступ к исходникам хоста | accept | Dev-only паттерн (hot-reload); в prod volume mount убирается, копия кода из COPY . . |
| T-03-04 | Spoofing | neo4j_password передаётся через env_file в docker-compose | accept | .env не в git (T-03-01 mitigated); Docker secrets для prod — за пределами MVP-фазы |
| T-03-05 | Denial of Service | connect_with_retry блокирует lifespan до 13s при недоступном Neo4j | accept | Задержки 1+3+9=13s допустимы при старте; graceful degradation не блокирует /health |
</threat_model>
