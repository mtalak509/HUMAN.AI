```python
from contextlib import asynccontextmanager
from functools import lru_cache

from fastapi import Depends, FastAPI, Request
from loguru import logger

# Импорт core-модулей (создаются в планах 01-02 и 01-03)
from core.config import Settings, get_settings as _get_settings
from core.graph import GraphDB
from core.logger import setup_logging


@asynccontextmanager
async def lifespan(app: FastAPI):
    # 1. Инициализировать Settings (per D-02)
    settings = _get_settings()
    app.state.settings = settings

    # 2. Настроить логирование один раз (per D-09, D-14)
    setup_logging(level=settings.log_level, json_mode=settings.log_json)
    logger.info("Logging configured: level={} json={}", settings.log_level, settings.log_json)

    # 3. Инициализировать GraphDB с retry (per D-04, D-05)
    db = GraphDB(uri=settings.neo4j_uri, user=settings.neo4j_user, password=settings.neo4j_password)
    await db.connect_with_retry(retries=3, delays=[1, 3, 9])
    app.state.db = db
    logger.info("Application startup complete")

    yield

    # Graceful shutdown
    await db.close()
    logger.info("Application shutdown complete")


app = FastAPI(title="HUMAN.AI", version="0.1.0", lifespan=lifespan)


# Dependency providers (per D-03) — читают из request.app.state
def get_settings(request: Request) -> Settings:
    return request.app.state.settings


def get_db(request: Request) -> GraphDB:
    return request.app.state.db


@app.get("/health")
async def health(db: GraphDB = Depends(get_db)) -> dict:
    """
    Всегда возвращает 200 OK (per D-06).
    Тело: {"status": "ok"} или {"status": "degraded", "neo4j": "unavailable"}.
    Docker health-check не падает при degraded.
    """
    if db is None or not db.is_connected:
        return {"status": "degraded", "neo4j": "unavailable"}
    try:
        await db.ping()
        return {"status": "ok"}
    except Exception:
        return {"status": "degraded", "neo4j": "unavailable"}
```

Важные детали реализации:
- Использовать `@asynccontextmanager` из contextlib (per D-01). Никаких `@app.on_event`.
- `lifespan` принимает `app: FastAPI` и передаётся в `FastAPI(lifespan=lifespan)`.
- `get_db` в роуте — отдельная функция для DI, не `_get_settings` из core.config.
- `db.connect_with_retry` — метод GraphDB (реализуется в плане 01-03). Здесь достаточно вызова.
- При `connect_with_retry` GraphDB устанавливает `self.is_connected = True/False`.
- Если GraphDB недоступен после всех ретраев — `app.state.db` всё равно устанавливается (degraded), `is_connected = False`.
- Не импортировать stdlib `get_settings` как публичный dependency provider — роуты используют локальные `get_settings(request)` и `get_db(request)`.
